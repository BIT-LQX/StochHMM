77  inline void assign(int16_t st, int16_t n_score, int16_t tb_ptr, int16_t tb_sc){
78  tb[(((int32_t) st) << 16 | n_score)] = (((int32_t)tb_ptr) << 16) | tb_sc;
79  }

81  inline void get(int16_t& st, int16_t& n_score){
82  int32_t key_val = (((int32_t) st) << 16 | n_score);
83  if (tb.count(key_val)){
84  st = -1;
85  }
86  int32_t val = tb[key_val];
87  st = (val >> 16) & 0xFFFF;
88  n_score = val & 0xFFFF;
89  }
