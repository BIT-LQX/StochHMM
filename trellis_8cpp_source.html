220  double trellis::getTransition(state* st, size_t trans_to_state, size_t sequencePosition){
221  double transition_prob(-INFINITY);
222  transition* trans = st->getTrans(trans_to_state);
223  if (trans==NULL){
224  return transition_prob;
225  }
226 
227  transType trans_type= trans->getTransitionType();
228 
229 
230 
231 
232  if (trans_type == STANDARD ){ //if the transition type is standard then just return the standard probability
233 
234  transition_prob= trans->getTransition(0,NULL);
235  }
236  else if (trans_type == DURATION){
237 
238  //If the traceback_table isn't defined, then we need to generate it using the viterbi algorithm
239  if (traceback_table==NULL){
240  (*this).viterbi();
241  }
242 
243  //Calculate the duration length from the traceback table
244  size_t size = get_explicit_duration_length(trans,sequencePosition, st->getIterator(), trans_to_state);
245  transition_prob=trans->getTransition(size,NULL);
246  }
247  else if (trans_type == LEXICAL){
248  transition_prob=trans->getTransition(sequencePosition, seqs);
249  }
250 
251  //Is external function define for the transition
252  if (trans->FunctionDefined()){
253 
254  //If the traceback_table isn't defined, then we need to generate it using the viterbi algorithm
255  if (traceback_table==NULL){
256  (*this).viterbi();
257  }
258 
259  transition_prob+=transitionFuncTraceback(st, sequencePosition, trans->getExtFunction());
260  }
261 
262  return transition_prob;
263  }

312  //! When a transitionFunc is to be called it must performs a traceback
313  //! and get the required sequence to pass to the function
314  //! \param
315  double trellis::transitionFuncTraceback(state* st, size_t position,transitionFuncParam* func){
316 
317  std::vector<int> tracebackPath;
318  std::vector<std::string> tracebackString;
319 
320  //How far to traceback
321  tracebackIdentifier traceback_identifier = func->getTracebackType();
322  const std::string& tracebackIdentifierName = func->getTracebackName();
323 
324 
325  //What to combine
326  combineIdentifier combineIdent = func->getCombineType();
327  const std::string& combineIdentName = func->getCombineName();
328 
329 
330  //Deterimine which track to use
331  track* alphaTrack = func->getTrack();
332  size_t trackIndex = alphaTrack->getIndex();
333  if (!alphaTrack->isAlpha()){
334 
335  std::cerr << "External transition function called on track that isn't discrete (ALPHANUMERIC)\n";
336  exit(2);
337 
338  }
339 
340  const sequence* seq = seqs->getSeq(trackIndex);
341  int16_t tb_state(st->getIterator());
342  int16_t starting_state = tb_state;
343 
344 
345  for(size_t trellisPos = position-1; trellisPos != SIZE_MAX ; --trellisPos){
346 
347  tracebackPath.push_back(tb_state);
348 
349  if ((combineIdent == FULL) ||
350  (combineIdent == STATENAME && combineIdentName.compare(st->getName())==0)||
351  (combineIdent == STATELABEL && combineIdentName.compare(st->getLabel())==0)||
352  (combineIdent == STATEGFF && combineIdentName.compare(st->getGFF())==0))
353 
354  {
355  tracebackString.push_back(seq->getSymbol(trellisPos));
356  }
357 
358  tb_state= (*traceback_table)[trellisPos][tb_state];
359  state* temp_st = hmm->getState(tb_state);
360 
361  //Check to see if stop conditions of traceback are met, if so break;
362  if(traceback_identifier == START_INIT && tb_state == -1) {break;}
363  else if (traceback_identifier == DIFF_STATE && starting_state != tb_state) { break;}
364  else if (traceback_identifier == STATE_NAME && tracebackIdentifierName.compare(temp_st->getName())==0){ break;}
365  else if (traceback_identifier == STATE_LABEL && tracebackIdentifierName.compare(temp_st->getLabel())==0) { break;}
366  else if (traceback_identifier == STATE_GFF && tracebackIdentifierName.compare(temp_st->getGFF())==0) { break;}
367  }
368 
369  size_t length=tracebackPath.size();
370  std::string CombinedString;
371 
372  size_t maxSymbolSize = alphaTrack->getAlphaMax();
373  //For single letter characters
374  if (maxSymbolSize ==1){
375  for(std::vector<std::string>::reverse_iterator rit = tracebackString.rbegin(); rit!=tracebackString.rend();++rit){
376  CombinedString+=(*rit);
377  }
378  }
379  else{ // For kmer words > 1 in length
380  std::vector<std::string>::reverse_iterator rit = tracebackString.rbegin();
381  CombinedString+=(*rit);
382  ++rit;
383  for(; rit!=tracebackString.rend();++rit){
384  CombinedString+="," + (*rit);
385  }
386  }
387 
388  //Call the transitionFunc and get the score back
389  double transitionValue = func->evaluate(seqs->getUndigitized(trackIndex), position, &CombinedString, length);
390 
391  return transitionValue;
392  }
